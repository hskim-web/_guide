/**
 * API
 * @class
 * @property {function} get GET 호출 방식
 * @property {function} post POST 호출 방식
 * @property {function} put PUT 호출 방식
 * @property {function} delete DELETE 호출 방식
 * @since 2022.01.11
 * @author 김희수
 */
const API = {
  /**
   * GET 방식
   * @memberof API
   * @namespace
   * @param {string} url 호스트를 포함한 URL 경로
   * @param {Object} [params={}] HTTP 요청 파라미터
   * @param {Object} [headers={}] HTTP 요청 헤더
   * @returns {Object} 응답 데이터
   * @since 2022.01.12
   * @author 김희수
   */
  get: async function (url, params = {}, headers = {}) {
    let query = Object.keys(params)
      .map(
        (key) => encodeURIComponent(key) + "=" + encodeURIComponent(params[key])
      )
      .join("&");
    let combinedUrl = `${url}?${query}`;
    let options = {
      method: "GET",
      headers: {
        "Content-Type": "application/json",
        ...headers,
      },
    };
    let res = await fetch(combinedUrl, options);
    let data = await res.json();

    if (res.ok) {
      return data;
    } else {
      throw Error(data);
    }
  },

  /**
   * POST 방식
   * @memberof API
   * @namespace
   * @param {string} url 호스트를 포함한 URL 경로
   * @param {Object} [params={}] HTTP 요청 파라미터
   * @param {Object} [headers={}] HTTP 요청 헤더
   * @returns {Object} 응답 데이터
   * @since 2022.01.12
   * @author 김희수
   */
  post: async function (url, params = {}, headers = {}) {
    let options = {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        ...headers,
      },
      body: JSON.stringify(params),
    };
    let res = await fetch(url, options);
    let data = await res.json();

    if (res.ok) {
      return data;
    } else {
      throw Error(data);
    }
  },

  /**
   * PUT 방식
   * @memberof API
   * @namespace
   * @param {string} url 호스트를 포함한 URL 경로
   * @param {Object} [params={}] HTTP 요청 파라미터
   * @param {Object} [headers={}] HTTP 요청 헤더
   * @returns {Object} 응답 데이터
   * @since 2022.01.12
   * @author 김희수
   */
  put: async function (url, params = {}, headers = {}) {
    let options = {
      method: "PUT",
      headers: {
        "Content-Type": "application/json",
        ...headers,
      },
      body: JSON.stringify(params),
    };
    let res = await fetch(url, options);
    let data = await res.json();

    if (res.ok) {
      return data;
    } else {
      throw Error(data);
    }
  },

  /**
   * DELETE 방식
   * @memberof API
   * @namespace
   * @param {string} url 호스트를 포함한 URL 경로
   * @returns {Object} 응답 데이터
   * @since 2022.01.12
   * @author 김희수
   */
  delete: async function (url) {
    let options = {
      method: "DELETE",
    };
    let res = await fetch(url, options);
    let data = await res.json();

    if (res.ok) {
      return data;
    } else {
      throw Error(data);
    }
  },
};
